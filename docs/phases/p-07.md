# ORGANIZATION MEMBERS

```sql id="2n1sgt"
CREATE TABLE organization_members (
    id UUID PRIMARY KEY,
    organization_id UUID,
    user_id UUID,
    role TEXT,
    joined_at TIMESTAMP DEFAULT NOW()
);
```

---

# API KEYS

```sql id="gsjlwm"
CREATE TABLE api_keys (
    id UUID PRIMARY KEY,
    organization_id UUID,
    key_hash TEXT,
    permissions JSONB,
    created_at TIMESTAMP DEFAULT NOW()
);
```

---

# AUDIT LOGS

```sql id="91khkz"
CREATE TABLE audit_logs (
    id UUID PRIMARY KEY,
    organization_id UUID,
    user_id UUID,
    action TEXT,
    metadata JSONB,
    created_at TIMESTAMP DEFAULT NOW()
);
```

---

# POLICIES

```sql id="6ytnkn"
CREATE TABLE policies (
    id UUID PRIMARY KEY,
    organization_id UUID,
    policy_type TEXT,
    configuration JSONB
);
```
