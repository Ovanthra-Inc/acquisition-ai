# 🧱 6. CORE CODE EXAMPLES

---

# 🔹 optimization_service.py

```python id="wcnzlj"
class OptimizationService:

    def optimize_campaign(self, metrics):
        recommendations = []

        if metrics["reply_rate"] < 5:
            recommendations.append(
                "Use more personalized subject lines"
            )

        if metrics["open_rate"] < 20:
            recommendations.append(
                "Test shorter email subjects"
            )

        return recommendations
```

class ExperimentationService:

    def create_ab_test(self, campaign_id, variants):
