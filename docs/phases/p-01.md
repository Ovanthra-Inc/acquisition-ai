## 🔹 models/lead.py

```python
from sqlalchemy import Column, String, Integer
from sqlalchemy.dialects.postgresql import UUID
import uuid
from app.db.base import Base

class Lead(Base):
    __tablename__ = "leads"

    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    user_id = Column(UUID(as_uuid=True))
    company_name = Column(String)
    website = Column(String)
    email = Column(String)
    score = Column(Integer, default=0)
```

---

## 🔹 repository/lead_repo.py

```python
from sqlalchemy.orm import Session
from app.models.lead import Lead

class LeadRepository:

    def create(self, db: Session, lead_data):
        lead = Lead(**lead_data.dict())
        db.add(lead)
        db.commit()
        db.refresh(lead)
        return lead

    def get_all(self, db: Session, user_id):
        return db.query(Lead).filter(Lead.user_id == user_id).all()
```

---

## 🔹 services/lead_service.py

```python
from app.repository.lead_repo import LeadRepository

class LeadService:

    def __init__(self):
        self.repo = LeadRepository()

    def create_lead(self, db, lead_data):
        return self.repo.create(db, lead_data)

    def list_leads(self, db, user_id):
        return self.repo.get_all(db, user_id)
```

---

## 🔹 api/v1/endpoints/leads.py

```python
from fastapi import APIRouter, Depends
from app.schemas.lead import LeadCreate, LeadResponse
from app.services.lead_service import LeadService
from app.db.session import get_db

router = APIRouter()
service = LeadService()

@router.post("/", response_model=LeadResponse)
def create_lead(data: LeadCreate, db=Depends(get_db)):
    return service.create_lead(db, data)

@router.get("/")
def list_leads(db=Depends(get_db)):
    return service.list_leads(db, user_id="dummy-user")
```
