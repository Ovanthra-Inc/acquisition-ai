# 🔹 reputation_service.py

```python id="r5s99x"
class ReputationService:

    def calculate_score(self, domain_metrics):
        score = 100

        if domain_metrics["bounce_rate"] > 5:
            score -= 30

        if domain_metrics["spam_rate"] > 2:
            score -= 40

        return max(score, 0)
```

---

# 🔹 rate_limiter.py

```python id="0ceo6c"
class RateLimiter:

    def allow_request(self, user_id, limit):
        current = redis.get(user_id)

        if current and int(current) >= limit:
            return False

        redis.incr(user_id)
        return True
```
