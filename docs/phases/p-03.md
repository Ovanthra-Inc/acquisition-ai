# 🧱 7. CORE CODE (SIMPLIFIED BUT REALISTIC)

---

## 🔹 planner.py

```python
class Planner:

    def create_plan(self, goal: str):
        if "saas" in goal.lower():
            return [
                "find_leads",
                "enrich_leads",
                "generate_email",
                "send_campaign"
            ]
        return []
```

---

## 🔹 executor.py

```python
class Executor:

    def execute(self, steps, context):
        results = {}

        for step in steps:
            if step == "find_leads":
                results["leads"] = search_leads(context)

            elif step == "enrich_leads":
                results["enriched"] = enrich(results["leads"])

            elif step == "generate_email":
                results["emails"] = generate_email(results["enriched"])

            elif step == "send_campaign":
                send_email(results["emails"])

        return results
```

---

## 🔹 agent_service.py

```python
class AgentService:

    def __init__(self):
        self.planner = Planner()
        self.executor = Executor()

    def run(self, goal):
        plan = self.planner.create_plan(goal)
        result = self.executor.execute(plan, {})
        return result
```
